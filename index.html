  • define python service interface
  • colander to specify message format
  • primary goal to reduce duplication and boilerplate


 class FoodLocationSchema(colander.MappingSchema):
 
     city = colander.SchemaNode(colander.String())
     state = colander.SchemaNode(colander.String())
     country = colander.SchemaNode(colander.String())
 
 
 class FoodTypesSchema(colander.MappingSchema):
 
     food_types = colander.SchemaNode(colander.Sequence(),
                                      colander.SchemaNode(colander.String()),
                                      description="List of food types.")
 
 food_types = http.GET(
         path='/location/food_types',
         request_schema=http.make_multi_schema(params=FoodLocationSchema(),
                                               headers=YelpHeaders),
         response_schema=http.make_multi_schema(body=FoodTypesSchema()))
